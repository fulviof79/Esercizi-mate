\begin{questions}
\question
\exonly{
Risolvere le seguenti disequazioni:
}

\begin{parts}
\part
\exonly{
$2x^4>0$
}
\solonly{
$S=]-\infty;0[ \cup ]0;\infty[$
}

\part
\exonly{
$x^4>1$
}
\solonly{
$S=]-\infty;-1[ \cup ]1;\infty[$
}
\part

\exonly{
$x^4<-x^2$
}
\solonly{
$S=\emptyset$
}

\part
\exonly{
$x^3\leq x^2 + 6x$
}
\solonly{
$S=]-\infty;-2] \cup [0;3]$
}

\part
\exonly{
$x^4 +7x^3>-10 x^2$
}
\solonly{
$S=]-\infty;-5[ \cup ]-2;0[ \cup ]0;\infty[$
}

\part

\exonly{
$x^3 +2x^2 -9x-18\geq0$
}
\solonly{
$S=[-3;-2] \cup [3;\infty[$
}

\end{parts}
